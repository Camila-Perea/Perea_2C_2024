137void Suministro_agua(void *pvParameter)
138{
139 uint16_t tiempo_cebado = 0;
140 uint16_t distancia;
141
142 while (1)
143 {
144 ulTaskNotifyTake(pdTRUE, portMAX_DELAY); // La tarea espera hasta recibir una notificación
145
146 if (!start)
147 {
148 noDispensarAgua(); // Apaga la bomba si el sistema no está habilitado
149 tiempo_cebado = 0; // Resetea el contador de cebado
150 continue; // Si no está habilitado, salta al siguiente ciclo
151 }
152
153 distancia = HcSr04ReadDistanceInCentimeters(); // Lee la distancia del sensor ultrasónico
154
155 if (distancia > 5 && distancia < 10) // Si el mate está en el rango de cebado
156 {
157 if (tiempo_cebado == 0) // Si es el primer ciclo de cebado
158 {
159 UartSendString(UART_PC, "Mate en rango, comienza a cebar\r\n");
160 }
161
162 if (tiempo_cebado < 5) // Si no ha alcanzado el tiempo máximo de cebado
163 {
164 dispensarAgua(); // Activa la bomba para cebar
165 tiempo_cebado++; // Aumenta el contador de cebado
166 }
167 else // Si ya se completó el cebado
168 {
169 noDispensarAgua(); // Desactiva la bomba
170 UartSendString(UART_PC, "Mate cebado, retírelo\r\n");
171 tiempo_cebado = 0; // Reinicia el contador para el siguiente cebado
172 }
173 }
174 else // Si el mate está fuera de rango
175 {
176 noDispensarAgua(); // Desactiva la bomba
177 tiempo_cebado = 0; // Resetea el contador de cebado
178 }
179 }
180}

186void Control_temperatura(void *pvParameter)
187{
188 uint16_t temperatura_mV;
189 uint16_t temperatura;
190 while (1)
191 {
192 ulTaskNotifyTake(pdTRUE, portMAX_DELAY); // La tarea espera en este punto hasta recibir una notificación
193
194 if (start)
195 {
196 AnalogInputReadSingle(SENSOR_TEMPERATURA, &temperatura_mV); // Lee el valor de la temepratura en voltaje (mV)
197 temperatura = ConvertirVoltajeATemperatura(temperatura_mV);
198
199 if (temperatura < TEMP_OBJEIVO)
200 {
201 calentarAgua();
202 UartSendString(UART_PC, " “Temperatura correcta…acerque el mate”\r\n"); // mensaje por uart
203 }
204 else
205 noCalentarAgua();
206 UartSendString(UART_PC, " “Agua fría…espere”\r\n"); // mensaje por uart
207 }
208 }
