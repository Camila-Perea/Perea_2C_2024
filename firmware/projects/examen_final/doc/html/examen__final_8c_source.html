89static void TareaControlAguaYAlimento(void *pvParameter)
90{
91 while (1)
92 {
93 ulTaskNotifyTake(pdTRUE, portMAX_DELAY);
94
95 if (sistema_on)
96 {
97 LedOn(LED_1);
98
99 /* ----------- CONTROL DE AGUA ----------- */
100 distancia_cm = HcSr04ReadDistanceInCentimeters();
101
102 if (distancia_cm > DISTANCIA_VACIO)
103 {
104 GPIOOn(GPIO_AGUA); // Falta agua, activo la electrovalvula
105 }
106 else if (distancia_cm < DISTANCIA_LLENO)
107 {
108 GPIOOff(GPIO_AGUA); // Recipiente lleno, desactivo la electrovalvula
109 }
110
111 /* ----------- CONTROL DE ALIMENTO ----------- */
112 AnalogInputReadSingle(CH0, &adc_alimento_mV);
113
114 // Conversion de mV a gramos
115 peso_alimento_g = (adc_alimento_mV * 1000) / 3300;
116
117 if (peso_alimento_g < 50)
118 {
119 GPIOOn(GPIO_ALIMENTO);
120 }
121 else if (peso_alimento_g > 500)
122 {
123 GPIOOff(GPIO_ALIMENTO);
124 }
125
126 // Conversion a cm3 para la lectura por uart
127 volumen_cm3 = 314 * (30 - distancia_cm);
128
129 /* ----------- INFORME POR UART ----------- */
130 UartSendString(UART_PC, "Agua: ");
131 UartSendString(UART_PC, (char *)UartItoa(volumen_cm3, 10));
132 UartSendString(UART_PC, " cm3, Alimento: ");
133 UartSendString(UART_PC, (char *)UartItoa(peso_alimento_g, 10));
134 UartSendString(UART_PC, " gr\r\n");
135 }
136 else
137 {
138 LedOff(LED_1);
139 GPIOOff(GPIO_AGUA);
140 GPIOOff(GPIO_ALIMENTO);
141 }
142
143 }
144}
