152void medirTemperatura(void *pvParameter)
153{
154 uint16_t temperatura_mV;
155 uint16_t temperatura;
156
157 int numMediciones = 10;
158 float mediciones[numMediciones];
159 float suma = 0.0;
160
161 while (1)
162 {
163 ulTaskNotifyTake(pdTRUE, portMAX_DELAY);
164
165 if (distancia >=8 && distancia <=12)
166 {
167 AnalogInputReadSingle(SENSOR_TEMPERATURA, &temperatura_mV);
168 temperatura = ConvertirVoltajeATemperatura(temperatura_mV);
169
170 for(int i = 0; i < numMediciones; i++)
171 {
172 mediciones[i] = temperatura;
173 suma += mediciones[i];
174 float promedio = suma / numMediciones;
175 }
176
177 // MESAJES POR LA UART
178 UartSendString(UART_PC, (char *) UartItoa(promedio, 10));
179 UartSendString(UART_PC, "Cº");
180 UartSendString(UART_PC, (char *) UartItoa(distancia, 10));
181 UartSendString(UART_PC, "cm\r\n");
182
183 if (temperatura < TEMP_OBJEIVO )
184 {
185 desactivarAlarma();
186 }
187 else if (temperatura > TEMP_OBJEIVO)
188 {
189 activarAlarma();
190 }
191 }
192 }
