189void Buzzer(void *pvParameter)
190{
191static bool estado = false;
192
193 while(1)
194 {
195 ulTaskNotifyTake(pdTRUE, portMAX_DELAY);
196
197 if(distancia >= 3000 && distancia < 5000)
198 {
199 // Precaución: toggle cada 1 s
200 estado = !estado;
201 if(estado)
202 BuzzerOn();
203 else
204 BuzzerOff();
205 }
206 else if(distancia < 3000)
207 {
208 // Peligro: toggle cada 0.5 s
209 estado = !estado;
210 if(estado)
211 BuzzerOn();
212 else
213 BuzzerOff();
214 }
215 else
216 {
217 // Distancia segura
218 estado = false;
219 BuzzerOff();
220 }
221 }

223void Acelerometro(void *pvParameter)
224{
225 while(1)
226 {
227 ulTaskNotifyTake(pdTRUE, portMAX_DELAY);
228 if (on == true)
229 {
230 AnalogInputReadSingle(CH1, &valor_sensado);
231 data_X = (valor_sensado - 1.65)/300; //convierto Voltaje a G
232 AnalogInputReadSingle(CH2, &valor_sensado);
233 data_Y = (valor_sensado - 1.65)/300; //convierto Voltaje a G
234 AnalogInputReadSingle(CH3, &valor_sensado);
235 data_Z = (valor_sensado - 1.65)/300; //convierto Voltaje a G
236
237 suma = data_X + data_Y + data_Z;
238 }
239
240 if(suma > 4)
241 {
242 MostrarMensaje(3);
243 }
244
245 }
